1.2 a = ++ a + 2 - --a - 3 + a ++ + 4
Result: a = 6+2-5-3+6+4 = 10

1.3 a = ++ a + ++ a + ++ a + a-- -6
Result: a = 6+7+8+7-6 = 22

1.4 a = a ++ + a + a-- + a-- + - a + a ++ - --a + a ++ - a-- + a - --a + a ++ + ++ a
